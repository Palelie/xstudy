    # 分巧克力

    标签：二分优化枚举

    # 题目描述

    # 输入描述

    # 输出描述

    输出切出的正方形巧克力最大可能的边长。

    # 输入输出样例

    # 示例

    输入

    2 10
6 5
5 6
    1
    2
    3

    输出

    2
    1

    # 运行限制

    • 最大运行时间：1s
    • 最大运行内存: 256M

    # 思路

    以图像的中心点出发，沿着实线对称前进实现图像分割。

    建立坐标系：以左下角为原点，出发点为（3，3）

    # 代码

    法一：暴力枚举（会超时间限制）

    #include <iostream>
using namespace std;
int main()
{
 //n为巧克巧克力数量，k为需要分给的小孩子数量
  int n,k;
 //记录每块巧克力的长和宽
  int h[100000];
  int w[100000];
  cin>>n>>k;
  for(int i=0;i<n;i++){
    cin>>h[i]>>w[i];
  }
  
  int len=100000;		//切割长度
  for(;len>=1;len--){
    int cnt=0;
//一共有n块巧克力， 每个巧克力块，都按照长度为len来切割
    for(int i=0;i<n;i++){
      cnt+=(h[i]/len)*(w[i]/len);
    }
    if(cnt>=k){
      cout<<len;
      return 0;
    }
  }
  return 0;
}

    法二：用二分法优化暴力枚举

    #include <iostream>
using namespace std;
int main()
{
 //n为巧克巧克力数量，k为需要分给的小孩子数量
  int n,k;
 //记录每块巧克力的长和宽
  int h[100000];
  int w[100000];
  cin>>n>>k;
  for(int i=0;i<n;i++){
    cin>>h[i]>>w[i];
  }
  
  int r=100001;
  int l=1;
  int ans=0;
//二分法优化枚举
  while(l<=r){
    int mid=(l+r)/2;
    int cnt=0;
//一共有n块巧克力， 每个巧克力块，都按照长度为mid来切割
    for(int i=0;i<n;i++){
      cnt+=(h[i]/mid)*(w[i]/mid);
    }
    if(cnt>=k){
      l=mid+1;
      ans=mid;   //每次符合的情况都记录下来，最后一次的结果便是所求结果
    }else{
        r=mid-1;
    }
  }
    cout<<ans;
  return 0;
}
