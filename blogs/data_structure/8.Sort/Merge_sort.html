    # 归并排序

    # 一、基本思想

    # 2-路归并排序：每次对两个子集进行排序，直到只有一个集合为止

    # 由于长得像一棵倒着的树，所以也称为“归并树”

    # 二、排序步骤

    # 三、算法分析

    # 解析：

    # 时间效率：一共需要log2n趟（归并树的深度），每一趟都需要归并n个数

    # 四、算法实现

    #include<iostream>
using namespace std;

void merge(int num[],int l,int r){
	//分配临时数组进行存储
	int *term = new int[r-l+1];
	int i = l,mid = (l + r) / 2;
	int j = mid+1;
	int x = 0;
	//归并
	while(i <= mid && j <= r){
		if(num[i] < num[j]) term[x++] = num[i++];
		else term[x++] = num[j++];
	}
	while(i <= mid) term[x++] = num[i++];
	while(j <= r) term[x++] = num[j++];
	//将临时数组数据转移回原数组
	for(int k = 0;k < x;k++) num[l++] = term[k];
	delete []term;
}

void merge_sort(int num[],int l,int r){
	if(l < r){
		int mid = (l + r) / 2;
		//分
		merge_sort(num,l,mid);
		merge_sort(num,mid+1,r);
		//治
		merge(num,l,r);
	}
}

int main(){
	int n;
	cout<<"请输入数字个数：";
	cin>>n;
	int num[n];
	cout<<"请输入该序列：";
	for(int i = 0;i < n;i++) cin>>num[i];
	merge_sort(num,0,n-1);
	cout<<"排序后序列：";
	for(int j = 0;j < n;j++) cout<<num[j]<<' ';
}
