    #include<iostream>
using namespace std;

//3分的希尔排序算法 
void shellsort(int* l,int len){
	for(int i=3;i>0;i--) shellInsert(l,i,len);		//循环1、2、3为希尔增量 
}

//进行单趟增量序列为k的希尔排序 
void shellInsert(int* l,int k,int len){
	int j; 
	for(int i=k+1;i<=len;i++){						//对k之后的每个数进行间隔为d的简单插入排序 
		if(l[i]<l[i-k]){						//由于前面的序列已经是有序的，因此若大于前一位数字，则不需要再进行比较 
			l[0]=l[i];
			for(j=i-k;j>0&&l[j]>l[0];j-=k) l[j+k]=l[j]; 	//单个数字间隔为d的简单插入排序 
			l[j+k]=l[0];
		}
	}
	cout<<"d="<<k<<"排序后的数组为：";
	for(int i=1;i<=len;i++) cout<<l[i]<<' ';		//输出排序后的数组
	cout<<endl;
}

int main(){
	int l[100],len;
	cout<<"请输入要排序的数字数量：";
	cin>>len;
	cout<<"请输入要排序的数字：";
	for(int i=1;i<=len;i++) cin>>l[i];		//输入原始数组 
	shellsort(l,len);
	return 0; 
}
